function cargarStock() {
            const tbody = document.getElementById('tbody-stock');
            tbody.innerHTML = '<tr><td colspan="6" class="text-center">⏳ Cargando datos...</td></tr>';

            try {
                const res = await fetch('/api/produccion/mantenimiento/stock');
                if (!res.ok) throw new Error('Error al cargar stock');
                const data = await res.json();
                
                window.mantenimientoStockData = data;
                aplicarFiltrosStock();

            } catch (error) {
                console.error(error);
                tbody.innerHTML = `<tr><td colspan="6" class="mensaje-error">❌ Error: ${error.message}</td></tr>`;
            }
        }
        
        function toggleFilterAccordion(id) {
            const content = document.getElementById(id);
            const icon = document.getElementById(id + '-icon');
            if (content.style.display === 'none') {
                content.style.display = 'block';
                icon.textContent = '▼';
            } else {
                content.style.display = 'none';
                icon.textContent = '▶';
            }
        }
